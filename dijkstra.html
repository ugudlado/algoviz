dist[source] = 0 pq = [(0, source)] while pq is not empty: (d, u) = pq.extractMin() if d > dist[u]: continue for (v, w) in adj[u]: if dist[u] + w < dist[v]: dist[v] = dist[u] + w prev[v] = u pq.insert((dist[v], v))
